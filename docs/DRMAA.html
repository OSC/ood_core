def DRMAA.version
    err= " " * ErrSize
    major = FFI::MemoryPointer.new(:int, 1)
    minor = FFI::MemoryPointer.new(:int, 1)
    r = FFI_DRMAA.drmaa_version major,minor, err, ErrSize
    r1 = [major.read_int,minor.read_int, err, ErrSize]  
    DRMAA.throw(r, r1[2])
    @version = r1[0] + (Float(r1[1])/100)
end
