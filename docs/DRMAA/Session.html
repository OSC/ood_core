def retry_until
    if @retry == 0
        job = yield
    else
        begin
            job = yield
        rescue DRMAA::DRMAATryLater
            STDERR.puts "... sleeping"
            sleep @retry
            retry
        end
    end
    return job
end

def wait_each(timeout = -1)
    if ! block_given?
        ary = Array.new
    end
    while true
        begin
            info = DRMAA.wait(ANY_JOB, timeout)
        rescue DRMAAInvalidJobError
            break
        end
        if block_given?
            yield info
        else
            ary << info
        end
    end
    if ! block_given?
        return ary
    end
end
