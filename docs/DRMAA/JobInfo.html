def initialize(job, stat, rusage = nil)
    @job = job
    @stat = stat.read_int
    @rusage = Hash.new

    if ! rusage.nil?
        DRMAA.get_attr_values(rusage).each { |u|
            nv = u.scan(/[^=][^=]*/)
            @rusage[nv[0]] = nv[1]
        }
    end
end
