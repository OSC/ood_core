def initialize(**opts)
  orig_name = opts.fetch(:name, 'unknown')
  @name = upcase_accounts? ? orig_name.upcase : orig_name
  @qos = opts.fetch(:qos, [])
  @cluster = opts.fetch(:cluster, nil)
  @queue = opts.fetch(:queue, nil)
end
